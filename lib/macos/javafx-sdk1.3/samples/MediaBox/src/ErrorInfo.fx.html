package com.sun.javafx.mediabox.controls;

import javafx.scene.control.Control;
import javafx.scene.control.Skin;
import javafx.scene.control.Labeled;
import javafx.scene.media.MediaError;
import com.sun.javafx.mediabox.MediaBox;
import com.sun.mediabox.controls.skin.Theme;

/**
 * @author baechul
 */

public class ErrorInfo extends Control, Labeled {

    public var x: Number = 0;
    public var y: Number = 0;
    public override var width = 320;
    public override var height = 180;

    public override var text = "MediaBox: We're sorry, this video can't be played now.";

    public var mediaError: MediaError on replace {
        if(mediaError != null) {
            // TODO: update error message based on an error type.
            show = true;
        } else {
            show = false;
        }
    }

    public var show: Boolean = false;

    init {
        visible = false;
    }

    package override function createDefaultSkin():Skin {
        Theme.getTheme(MediaBox.theme).createErrorInfoSkin();
    }
}
